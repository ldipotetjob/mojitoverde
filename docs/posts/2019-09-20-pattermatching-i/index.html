Pattern Sequences:

SimplePattern ::= StableId ‘(’ [Patterns ‘,’] [varid ‘@’] ‘_’ ‘*’ ‘)’

 1val testList:List[Int] = List(1,1)
 2
 3testList match {
 4  case List(1, _*) => s"hello $testList"
 5  case _ => " nothing match"
 6}
 7
 8// res23: String = hello List(1, 1)
 9
10val testListbinding:List[Int] = List(1,1)
11
12testList match {
13  case testList@List(1, _*) => "a list beginning with 1, binding"
14  case _ => " nothing match  binding"
15}
16
17// testListbinding: List[Int] = List(1, 1)
18// res24: String = a list beginning with 1, binding
19
20testList match {
21  case x @ List(1, _*) => s"$x"
22  case _ => " nothing match  binding"
23}
24
25// res25: String = List(1, 1)

Pattern Alternatives:

Pattern ::= Pattern1 { ‘|’ Pattern1 }

 1def matchDifferentPatterns(x: Any): Any = x match {
 2  case 1 | 2 => "one or two"
 3  case "two" => 2
 4  case 3 | 4 => "three or four"  
 5  case y: Int => "scala.Int"
 6  case _ => "many"
 7}
 8
 9matchDifferentPatterns(1) // res37: Any = one or two
10
11matchDifferentPatterns(2) // res38: Any = one or two
12
13matchDifferentPatterns(3) // res39: Any = three or four
14
15matchDifferentPatterns(4) // res40: Any = three or four
